#importing libraries
import bpy
import math
from mathutils import Vector

#selecting a project object
esf=bpy.data.objects['Corpo_Padrao']

#select collection (only on the main level)
collections = bpy.context.view_layer.layer_collection.children

for collection in collections:
    if collection.name == "Caminho":
        bpy.context.view_layer.active_layer_collection = collection

#adjusting initial parameters
framein=150
x=0
y=x+3
anginicial=-180
angfinal=360
anginicialobj=0

#make a new curve
crv = bpy.data.curves.new('crv', 'CURVE')
crv.dimensions = '3D'

#make a new spline in that curve
spline = crv.splines.new(type='NURBS')

#positioning the object, creating keyframes and spline points
while x<=angfinal:
    theta=((anginicial+x)*math.pi)/180
    coord=[x/180,1.5*math.cos(theta),1.5*math.sin(theta)]
    esf.location=coord
    esf.keyframe_insert(data_path = "location", frame=framein+x+1)
    esf.rotation_mode = "XYZ"
    esf.rotation_euler.y = math.radians(anginicialobj+x)
    esf.keyframe_insert(data_path = 'rotation_euler',
    frame = framein+x+1)
    spline.points.add(1)
    spline.points[-1].co = [coord[0],coord[1],coord[2],1]
    x=x+1
obj = bpy.data.objects.new('cube', crv)
bpy.context.scene.collection.objects.link(obj)

void selecionar_palavra (char *palavrasort,int nivel)
{
    char nv[3][15]={"colher","bragantino","chikungunya"};
    int i;

    //copiando a string do nivel atual
    for (i=0;nv[nivel][i]!='\0';i++)
    {
        palavrasort[i]=nv[nivel][i];
    }
    palavrasort[i]='\0';
